        private static TVar _elu(TVar input)
        {
            float alpha = 1;
            var keepElements = input >= 0;
            TVar d = (alpha * (input.Exp() - 1)).Evaluate();
            return (input.CMul(keepElements) + (1 - keepElements).CMul(d)).Evaluate();
        }
