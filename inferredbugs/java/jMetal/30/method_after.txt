  @Override
  public int compare(Solution solution1, Solution solution2) {
    if (solution1 == null) {
      return 1;
    } else if (solution2 == null) {
      return -1;
    }

    if (NSGAIIAttr.getAttributes(solution1).getRank() < NSGAIIAttr.getAttributes(solution2).getRank()) {
      return -1;
    }

    if (NSGAIIAttr.getAttributes(solution1).getRank() > NSGAIIAttr.getAttributes(solution2).getRank()) {
      return 1;
    }

  }
