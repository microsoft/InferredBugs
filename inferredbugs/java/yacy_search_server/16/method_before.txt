    public static byte[] read(File source) throws IOException {
        byte[] buffer = new byte[(int) source.length()];
        InputStream fis = new FileInputStream(source);
        int p = 0;
        int c;
	while ((c = fis.read(buffer, p, buffer.length - p)) > 0) p += c;
	fis.close();
        return buffer;
    }
