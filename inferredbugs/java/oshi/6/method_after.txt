 public long getTotal() {
		if (totalMemory == 0) {
			Sysinfo info = new Sysinfo();
			if (0 != Libc.INSTANCE.sysinfo(info))
				throw new LastErrorException("Error code: "
						+ Native.getLastError());
			totalMemory = info.totalram.longValue() * info.mem_unit;
		}
		return totalMemory;
	}
