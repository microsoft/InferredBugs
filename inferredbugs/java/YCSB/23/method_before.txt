  @Override
  public boolean doTransaction(DB db, Object threadstate) {
    switch (operationchooser.nextString()) {
    case "UPDATE":
      doTransactionUpdate(db);
      break;
    case "INSERT":
      doTransactionInsert(db);
      break;
    case "DELETE":
      doTransactionDelete(db);
      break;
    default:
      doTransactionRead(db);
    }
    return true;
  }
